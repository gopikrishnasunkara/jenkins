- hosts: hosts
  remote_user: ubuntu
  become: True
  tasks:
    - name: Install nginx
      apt:
        name: nginx
        state: present
        update_cache: yes
        cache_valid_time: 3600
    - name: Create default index page
      copy:
        src: /etc/ansible/index.html
        dest: /usr/share/nginx/html/index.html
	remote_src: yes
    - name: Restart nginx
      service:
        name: nginx
        state: restarted
